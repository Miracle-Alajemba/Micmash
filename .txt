Liking Logic	EventLikeController.php

3. I want to change DATA settings (Database)
If you want to change...	Go to file...

Allowed columns (Fix "Mass Assignment" error)	app/Models/Event.php (Look for $fillable)
Add a new Column (e.g., adding "Price" to events)	database/migrations/ (You must create a new migration)
Database Password / Name	.env file (at the very bottom of your project)

If you want to change...	Go to file...
The link address (e.g., changing /events to /activities)	routes/web.php
Middleware (Who is allowed to see what)	routes/web.php

// 4. Store New Event
    public function store(Request $request)
    {
        $validated = $request->validate([
            'title'       => 'required|string|max:255',
            'description' => 'required|string',
            'location'    => 'required|string',
            'date'        => 'required|date|after:today',
            'time'        => 'required',
            'category_id' => 'required|exists:event_categories,id',
            'image'       => ['required', File::types(['jpg', 'jpeg', 'webp', 'png'])->max(2048)],
            'url'         => 'nullable|active_url',
            // Speaker validation...
            'speakers'    => 'nullable|array',
            'speakers.*.name'  => 'nullable|string|max:255',
            'speakers.*.role'  => 'nullable|string|max:255',
            'speakers.*.image' => 'nullable|image|max:2048',
        ]);

        if ($request->hasFile('image')) {
            $file = $request->file('image');
            $filename = time() . '_' . $file->getClientOriginalName();

            // âœ… SAVING TO 'eventimages' FOLDER ON PUBLIC DISK
            $file->storeAs('eventimages', $filename, 'public');

            // âœ… SAVING FILENAME TO DATABASE
            $validated['image'] = $filename;
        }

        $validated['user_id'] = Auth::id();
        $validated['status'] = 'pending';

        $event = Event::create($validated);

        // Speakers logic (remains the same)...
        if ($request->has('speakers')) {
            foreach ($request->speakers as $index => $speakerData) {
                if (empty($speakerData['name'])) continue;
                $speakerImageName = null;
                if ($request->hasFile("speakers.{$index}.image")) {
                    $file = $request->file("speakers.{$index}.image");
                    $speakerImageName = time() . '_spk_' . $index . '_' . $file->getClientOriginalName();
                    $file->storeAs('speakers', $speakerImageName, 'public');
                }
                EventSpeaker::create([
                    'event_id' => $event->id,
                    'name'     => $speakerData['name'],
                    'role'     => $speakerData['role'] ?? null,
                    'image'    => $speakerImageName,
                ]);
            }
        }

        return redirect()->route('events.index')->with('success', 'Event created successfully!');
    }

    // 6. Update Event
    public function update(Request $request, Event $event)
    {
        if (Auth::id() !== $event->user_id && !Auth::user()->is_admin) {
            abort(403, 'Unauthorized action.');
        }

        $validated = $request->validate([
            'title'       => 'required|string|max:255',
            'description' => 'required',
            'location'    => 'required',
            'date'        => 'required|date',
            'time'        => 'required',
            'category_id' => 'required|exists:event_categories,id',
            'image'       => 'nullable|image|max:2048'
        ]);

        if ($request->hasFile('image')) {
            // âœ… DELETE OLD IMAGE FROM 'eventimages'
            if ($event->image) {
                Storage::disk('public')->delete('eventimages/' . $event->image);
            }

            $file = $request->file('image');
            $filename = time() . '_' . $file->getClientOriginalName();

            // âœ… SAVE NEW IMAGE TO 'eventimages'
            $file->storeAs('eventimages', $filename, 'public');

            $validated['image'] = $filename;
        }

        $event->update($validated);

        return redirect()->route('events.show', $event)->with('success', 'Event updated successfully.');
    }
